Replaces all of the models in the collection with the input array

item
A model to replace all of the models in the current collection with.
items
An array of models to replace the current list of models in the collection.
{% highlight javascript linenos %} var Dog = Falcon.Model.extend(); var Dogs = Falcon.Collection.extend({model: Dog}); var dogs = new Dogs([ {"name": "Fido"}, {"name": "Skip"} ]); console.log( dogs.pluck("name") ); // [Fido, Skip] dogs.replace({"name": "Lassie"}); console.log( dogs.pluck("name") ); // [Lassie] dogs.replace([{"name": "Bucko"}, ["name": "Spot"]]); console.log( dogs.pluck("name") ); // [Bucko, Spot] {% endhighlight %}
