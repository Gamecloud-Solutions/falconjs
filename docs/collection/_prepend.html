Adds a model or an array of models to the beginning of this collection.

item
A model to prepend to the beginning of this collection.
items
An array of models to prepend to the beginning of this collection.
{% highlight javascript linenos %} var Dog = Falcon.Model.extend(); var Dogs = Falcon.Collection.extend({model: Dog}); var dogs = new Dogs(); dogs.prepend({"name": "Fido"}); console.log( dogs.pluck("name") ); // => [Fido] dogs.prepend( new Dog({name: "Spot"}) ); console.log( dogs.pluck("name") ); // => [Spot, Fido] dogs.prepend([ {name: "Skip"}, {name: "Bump"} ]); console.log( dogs.pluck("name") ); // => [Skip, Bump, Spot, Fido] {% endhighlight %}
