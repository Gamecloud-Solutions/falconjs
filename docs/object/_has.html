Checks to see if a specific event is defined for an object. Optionally, if a callback is given, this method will check to see if that callback is defined for the specified event.

event
The event to check.
callback
A specific callback for an event to check.
{% highlight javascript linenos %} var obj = new Falcon.Object(); var log_function = function() { log("Hello World") }; var foo_function = function() { log("Foo Bar") }; obj.on("log", log_function); console.log( obj.has("log") ); // => true console.log( obj.has("click") ); // => false console.log( obj.has("log", log_function) ); // => true console.log( obj.has("log", foo_function) ); // => false {% endhighlight %}
