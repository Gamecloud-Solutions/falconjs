Removes event listening for a specific event. Optionally, if a callback is given, only that callback will be removed from the list of callbacks for the specified event.

event
The event to stop listening for.
callback
The specific callback function to remove.
{% highlight javascript linenos %} var obj = new Falcon.Object(); obj.on("log", function() { console.log("Hello World"); }); obj.trigger("log"); // => Logs 'Hello World' obj.off("log"); obj.trigger("log"); // => Nothing Happens {% endhighlight %} {% highlight javascript linenos %} var obj = new Falcon.Object(); var log_callback_one = function() { console.log("Log One"); }; var log_callback_two = function() { console.log("Log Two"); }; obj.on("log", log_callback_one); obj.on("log", log_callback_two); obj.trigger("log"); // => Logs 'Log One', then 'Log Two' obj.off("log", log_callback_one); obj.trigger("log"); // => Logs 'Log Two' {% endhighlight %}
