The url or element id where this template is defined. Can either be a string or a function and can either point to a uri or a DOM object identifier. This attribute should be overridden by any inheriting classes

{% highlight javascript linenos %} var DogUrlView = Falcon.View.extend({ url: "dog.tmpl" }); var DogIdView = Falcon.View.extend({ url: "#dog" }); var DogFunctionView = Falcon.View.extend({ url: function() { return "dog.html" } }); var dog_url_view = new DogUrlView(); var dog_id_view = new DogIdView(); var dog_function_view = new DogFunctionView(); Falcon.baseTemplateUrl = "http://www.falconjs.com/"; console.log( dog_url_view.makeUrl() ); // => http://www.falconjs.com/dog.tmpl console.log( dog_id_view.makeUrl() ); // => #dog console.log( dog_function_view.makeUrl() ); // => http://www.falconjs.com/dog.html {% endhighlight %}
