This is a hash map of the default observables and values that will become properties of this view when a new instance is created. If the value of a specific key is a function a computed observable is created. If the value is an object with the keys 'read' and/or 'write' a computed is created with those object keys as the definition for the computed observable. All computed observables are created with this objects's instance as their context.

{% highlight javascript linenos %} var DogView = Falcon.View.extend({ observables: { "name": "Fido", "parent_dog_name": "", "child_of_lassie": function(parent_dog_name) { return this.parent_dog_name() === "Lassie"; } } }); var dog_view = new DogView(); // Dog not related to Lassie console.log( ko.isObservable( dog_view.name ) ); // => true console.log( ko.isComputed( dog_view.child_of_lassie ) ); // => true console.log( dog_view.name() ); // => Fido console.log( dog_view.child_of_lassie() ); // => false dog_view.name("Spot"); dog_view.parent_dog_name("Lassie"); console.log( dog_view.name() ); // => Spot console.log( dog_view.child_of_lassie() ); // => true {% endhighlight %}
