Executed when this view is displayed (added to the DOM). This method is called from the view binding when rendering this view's template is complete and added to the DOM. This should be overridden by inheriting classes and should primarily be used for 'on-display' functionality such as certain event handlers

{% highlight javascript linenos %} var DogView = Falcon.View.extend({ url: '#dog_view', display: function() { console.log("Dog View Displayed") } }); var view_model = { showing_dog_view: ko.observable(false), my_dog_view: new DogView() }; console.log("Initializing"); setTimeout(function() { view_model.show_dog_view(true) }, 5000); Falcon.apply(view_model, '#application'); {% endhighlight %}
