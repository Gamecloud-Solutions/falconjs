--- title: shift method: collection.shift() returnType: Falcon.Model returnDescription: The first model in this collection complete: false category: collection ---

Removes the first element from the collection and returns it.

var Dog = Falcon.Model.extend(); var Dogs = Falcon.Collection.extend({ model: Dog }); var dogs = new Dogs([ {"name": "Fido"}, {"name": "Skip"}, {"name": "Bump"}, {"name": "Bucko"} ]); var first_dog = dogs.shift(); console.log( dogs.length() ); // => 3 console.log( first_dog.get("name") ); // => Fido
