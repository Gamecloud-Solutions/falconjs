--- title: models method: collection.models complete: false category: collection ---

The internal observable array that contains all of the models added to this collection.

var Dog = Falcon.Model.extend(); var Dogs = Falcon.Collection.extend({ model: Dog }); var dogs = new Dogs([ {"name": "Fido"}, {"name": "Skip"}, {"name": "Bump"}, {"name": "Bucko"} ]); console.log( ko.isObservable( dogs.models ) ); // => true console.log( dogs.models() ); // => [Dog, Dog, Dog, Dog]
