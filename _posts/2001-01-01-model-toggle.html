--- title: toggle method: model.toggle( attribute ) returnType: Falcon.Model returnDescription: This instance. complete: false category: model ---

Toggles the value between true/false on the specific attribute of this model.

attribute
The attribute in the model to look up.
var Dog = Falcon.Model.extend({ defaults: { "is_tame": true } }); var dog = new Dog({"is_tame": true}); console.log( dog.get("is_tame") ); // => true dog.toggle("is_tame"); console.log( dog.get("is_tame") ); // => false
